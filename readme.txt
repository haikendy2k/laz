    private void btnsuaActionPerformed(java.awt.event.ActionEvent evt) {                                       
        int soDongChon = tblnhacungcap.getSelectedRowCount();
        if (soDongChon!=1) {
            GUI.thongBao.thongBao("Vui lòng chọn dòng để sửa", "Thông báo");
            return;
        }
        int dongDuocChon = tblnhacungcap.getSelectedRow();
        nhacungcapedit = tblnhacungcap.getValueAt(dongDuocChon, 0).toString();
        jdlsuanhacungcap jdl  = new jdlsuanhacungcap(this, true);
        jdl.setVisible(true);
        BLL.BLLnhacungcap.doDuLieuNhaCungCap(tblnhacungcap);
